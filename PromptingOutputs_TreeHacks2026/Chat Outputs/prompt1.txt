Generate an OpenSCAD script that creates a 3D model of a classic American fire hydrant matching the attached reference image (front-facing studio photo). The model should be clean, printable, and parameterized. VISUAL FEATURES TO MATCH (from the image) 1) Base: wide circular flange at the bottom with a thicker rim and bolt bumps (or simplified bolt heads) evenly spaced around. 2) Main body: vertical cylindrical column with subtle step changes (slight taper/shoulders) and a thicker mid-band. 3) Side nozzles: two symmetric side outlets (left and right), each a short cylindrical protrusion with a cap-like end. /**************
* PARAMETERS  *
**************/
$fn = 64;

// Global sizing
overall_h        = 200;   // mm
base_d           = 110;
body_d           = 66;
dome_d           = 82;
nozzle_d         = 38;
nozzle_len       = 46;
front_boss_d     = 28;

// Detailing / printability
min_feature      = 1.8;   // target minimum thickness
round_r          = 1.2;   // light rounding radius (keep small for printability)

// Base
base_h           = 18;
base_rim_h       = 8;
base_rim_over    = 8;     // rim overhang (radius increase from base body)
bolt_count       = 8;
bolt_d           = 10;
bolt_h           = 5;

// Body profile
body_h           = 132;   // height from base top to dome seam
shoulder_h       = 10;
neck_h           = 10;
mid_band_h       = 16;
mid_band_over    = 6;

// Nozzles placement
nozzle_z         = base_h + 62;
nozzle_y_offset  = 0;     // keep centered in Y (front is +Y)
nozzle_lug_w     = 10;
nozzle_lug_h     = 8;
nozzle_lug_t     = 6;

// Front boss
front_boss_z     = base_h + 60;
front_boss_out   = 6;     // how far it sticks out
front_nub_len    = 7;
front_nub_d1     = 12;    // base
front_nub_d2     = 6;     // tip

// Dome
dome_h           = overall_h - (base_h + body_h); // auto
dome_seam_h      = 6;
dome_rib_count   = 12;
dome_rib_depth   = 2.2;
dome_rib_w       = 6;     // groove width (cut)
top_knob_d       = 16;
top_knob_h       = 10;
top_knob_flat    = 6;

// Chains (optional)
include_chains   = true;
chain_r          = 2.2;   // tube radius for simplified chain
chain_sag        = 16;    // sag depth
chain_samples    = 9;     // number of segments per chain

/****************
* HELPER UTILS  *
****************/
function vlerp(a,b,t) = a + (b-a)*t;
function bez2(p0,p1,p2,t) =
    let(u=1-t)
    p0*u*u + p1*2*u*t + p2*t*t;

module rounded_cylinder(h=10, d=20, r=1.0, center=false) {
    // Lightly rounded vertical cylinder by minkowski (kept small)
    // Note: expands height by 2r; we compensate by shrinking core height.
    minkowski() {
        cylinder(h=max(0.01, h-2*r), d=max(0.01, d-2*r), center=center);
        sphere(r=r);
    }
}

module tube_from_points(pts, r=2.0) {
    // Manifold tube via hull of spheres between successive points
    for(i=[0:len(pts)-2]) {
        hull() {
            translate(pts[i])   sphere(r=r);
            translate(pts[i+1]) sphere(r=r);
        }
    }
}

/****************
* MODEL MODULES *
****************/
module base_flange() {
    // Base body + rim + bolts, with light rounding on rim and base edge.
    union() {
        // Main base disk
        difference() {
            union() {
                // Base disk with slight rounding
                rounded_cylinder(h=base_h, d=base_d, r=round_r);

                // Thick rim ring (slightly wider, on top portion)
                translate([0,0,base_h-base_rim_h])
                    rounded_cylinder(h=base_rim_h, d=base_d + 2*base_rim_over, r=round_r*0.9);
            }
            // Flatten bottom to Z=0 perfectly (minkowski can introduce tiny rounding below)
            translate([0,0,-2])
                cylinder(h=2, d=base_d + 2*(base_rim_over+10));
        }

        // Bolt bumps around the rim
        for(k=[0:bolt_count-1]) {
            a = 360/bolt_count * k;
            translate([ (base_d/2 + base_rim_over*0.6)*cos(a),
                        (base_d/2 + base_rim_over*0.6)*sin(a),
                        base_h - base_rim_h + 2 ])
                cylinder(h=bolt_h, d=bolt_d, center=false);
        }

        // Small step-up shoulder from base to body
        translate([0,0,base_h-2])
            cylinder(h=shoulder_h, d1=body_d+10, d2=body_d, center=false);
    }
}

module main_body() {
    // Stacked profile with subtle steps and mid-band.
    union() {
        z0 = base_h;

        // Lower body (slight taper)
        translate([0,0,z0])
            cylinder(h=50, d1=body_d+2, d2=body_d-1);

        // Mid-band (thicker ring)
        translate([0,0,z0+46])
            cylinder(h=mid_band_h, d=body_d + 2*mid_band_over);

        // Upper body
        translate([0,0,z0+46+mid_band_h])
            cylinder(h=body_h-(46+mid_band_h)-neck_h, d=body_d-1);

        // Neck to dome seam
        translate([0,0,z0+body_h-neck_h])
            cylinder(h=neck_h, d1=body_d-1, d2=dome_d-8);

        // Seam ring under dome
        translate([0,0,z0+body_h])
            cylinder(h=dome_seam_h, d=dome_d-8);
    }
}

module side_nozzle(sign=1) {
    // sign=+1 right, -1 left (along X axis)
    xdir = sign;

    // Nozzle centerline position
    nx = xdir*(body_d/2);
    nz = nozzle_z;
    ny = nozzle_y_offset;

    union() {
        // Nozzle barrel + cap end (aligned along X)
        translate([nx, ny, nz]) rotate([0,90,0]) {
            // barrel
            cylinder(h=nozzle_len, d=nozzle_d, center=false);
            // cap-like end
            translate([nozzle_len-6,0,0])
                cylinder(h=10, d1=nozzle_d+6, d2=nozzle_d+2, center=false);
            // small end boss
            translate([nozzle_len+3,0,0])
                sphere(d=nozzle_d*0.65);
        }

        // Nozzle root fillet (hull to body)
        hull() {
            translate([xdir*(body_d/2 - 2), ny, nz]) sphere(r=nozzle_d*0.30);
            translate([xdir*(body_d/2 + 8), ny, nz]) sphere(r=nozzle_d*0.30);
        }

        // Rectangular lug/boss on nozzle (seen in reference)
        // Place on outer side (toward +/-X), slightly above centerline
        lug_x = xdir*(body_d/2 + nozzle_len*0.58);
        lug_y = ny + nozzle_d*0.22; // toward front a bit
        lug_z = nz + nozzle_d*0.18;
        translate([lug_x, lug_y, lug_z])
            rotate([0,0,0])
                cube([nozzle_lug_t, nozzle_lug_w, nozzle_lug_h], center=true);
    }
}

module front_boss() {
    // Small circular boss and conical/rounded nub on front midsection.
    y = body_d/2 + front_boss_out;
    z = front_boss_z;

    union() {
        // Boss disk
        translate([0, y, z])
            rotate([90,0,0])
                cylinder(h=front_boss_out+2, d=front_boss_d, center=false);

        // Nub on boss (conical + rounded tip)
        translate([0, y + front_boss_out+1.2, z])
            rotate([90,0,0]) {
                cylinder(h=front_nub_len, d1=front_nub_d1, d2=front_nub_d2, center=false);
                translate([0,0,front_nub_len])
                    sphere(d=front_nub_d2);
            }

        // Slight blending into body
        hull() {
            translate([0, body_d/2 - 1, z]) sphere(r=front_boss_d*0.25);
            translate([0, y-1, z])         sphere(r=front_boss_d*0.25);
        }
    }
}

module dome_cap() {
    // Domed "helmet" with ribbing grooves and top knob.
    z0 = base_h + body_h;

    difference() {
        union() {
            // Dome base cylinder (skirt)
            translate([0,0,z0+dome_seam_h])
                cylinder(h=10, d=dome_d-4);

            // Main dome: intersection of sphere and cylinder-ish to keep printable silhouette
            translate([0,0,z0+dome_seam_h+6])
                intersection() {
                    // sphere provides dome curvature
                    translate([0,0,(dome_d*0.50)])
                        sphere(d=dome_d);
                    // keep lower half + limit max diameter
                    cylinder(h=dome_h+30, d=dome_d, center=false);
                }

            // Top knob (dark cap in photo)
            translate([0,0,overall_h - top_knob_h])
                union() {
                    cylinder(h=top_knob_h, d=top_knob_d);
                    // flattened top
                    translate([0,0,top_knob_h])
                        cylinder(h=top_knob_flat, d=top_knob_d*0.85);
                }
        }

        // Ribbing grooves: subtract shallow vertical channels around dome
        // (Cutting thin rotated boxes creates grooves; depth controlled by box intrusion.)
        for(i=[0:dome_rib_count-1]) {
            a = 360/dome_rib_count * i;
            rotate([0,0,a]) {
                // Place cutter around dome perimeter; tall enough to cover dome region
                translate([ (dome_d/2) - dome_rib_depth, 0, z0 + dome_seam_h + 2 ])
                    cube([ dome_rib_depth + 1.5, dome_rib_w, dome_h + 30 ], center=true);
            }
        }

        // Small seam indentation near where dome meets body (subtle step line)
        translate([0,0,z0 + dome_seam_h + 2])
            cylinder(h=2, d=dome_d-10);
    }
}

module chain_pair() {
    // Simplified draped chains (two tubes) from nozzle regions to front boss area
    // Uses quadratic Bezier with sag.
    if(include_chains) {
        // Anchor points (approximate)
        left_start  = [-(body_d/2 + nozzle_len*0.72),  nozzle_d*0.10, nozzle_z - 2];
        right_start = [ (body_d/2 + nozzle_len*0.72),  nozzle_d*0.10, nozzle_z - 2];

        front_end   = [0, body_d/2 + front_boss_out + 8, front_boss_z - 4];

        // Control points (sag down in Z and slightly forward)
        left_ctrl   = [-(body_d/2 + nozzle_len*0.25),  body_d*0.22, nozzle_z - chain_sag];
        right_ctrl  = [ (body_d/2 + nozzle_len*0.25),  body_d*0.22, nozzle_z - chain_sag];

        // Build sampled point lists
        ptsL = [ for(t=[0:1/(chain_samples-1):1]) bez2(left_start,  left_ctrl,  front_end, t) ];
        ptsR = [ for(t=[0:1/(chain_samples-1):1]) bez2(right_start, right_ctrl, front_end, t) ];

        // Tubes
        tube_from_points(ptsL, r=chain_r);
        tube_from_points(ptsR, r=chain_r);

        // Small end "hooks"/nubs (print-friendly)
        translate(left_start)  sphere(r=chain_r*1.3);
        translate(right_start) sphere(r=chain_r*1.3);
        translate(front_end)   sphere(r=chain_r*1.4);
    }
}

/****************
* FINAL ASSEMBLY *
****************/
module hydrant() {
    union() {
        base_flange();
        main_body();

        // Side nozzles
        side_nozzle(+1);
        side_nozzle(-1);

        // Front boss
        front_boss();

        // Dome cap
        dome_cap();

        // Chains (optional)
        chain_pair();
    }
}

// Render
hydrant();
